# 参数默认值
```js
//在 ES6 以前，我们无法给一个函数参数设置默认值，只能采用变通写法：
function add(a, b) {
  // 判断 b 是否为空，为空就给默认值 1
  b = b || 1;
  return a + b;
}
// 传一个参数
console.log(add(10));

//现在可以这么写：直接给参数写上默认值，没传就会自动使用默认值
function add2(a, b = 1) {
  return a + b;
}

// 传一个参数
console.log(add2(10));
```

# 解构赋值
- 数组解构
```js
let a = 1;
let b = 2;
let c = 3;

let [a, b, c] = [1, 2, 3];
```
- 对象解构
按属性取值
```js
let { bar, foo } = { foo: 'aaa', bar: 'bbb' };
foo // "aaa"
bar // "bbb"

let { baz } = { foo: 'aaa', bar: 'bbb' };
baz // undefined
```

参数默认值
```js
var {x = 3} = {};
x // 3

var {x, y = 5} = {x: 1};
x // 1
y // 5

var {x: y = 3} = {};
y // 3

var {x: y = 3} = {x: 5};
y // 5

var { message: msg = 'Something went wrong' } = {};
msg // "Something went wrong"
```

# Promise

```js
function fetchData() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      const success = true; // 模拟成功或失败
      if (success) {
        resolve('Data received');
      } else {
        reject('Failed to fetch data');
      }
    }, 2000);
  });
}

fetchData()
  .then((data) => {
    console.log(data); // "Data received"
  })
  .catch((error) => {
    console.error(error); // "Failed to fetch data"
  });

```
