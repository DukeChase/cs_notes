# ============================================================
# CS Notes - robots.txt
# 功能：控制搜索引擎爬虫对站点内容的索引
#
# 部署位置：
#   根路径部署 → /var/www/cs-notes/robots.txt
#   子路径部署 → /var/www/cs-notes/cs_notes/robots.txt
#
# 注意：robots.txt 只是「君子协定」，恶意爬虫不会遵守
#       敏感内容的真正保护需要依赖 Nginx 访问控制（见 nginx.conf）
# ============================================================

# -------------------- 全局规则 --------------------
User-agent: *
# 允许抓取公开内容
Allow: /
# 禁止抓取考试资料（sys-analyst、sys-architect）
Disallow: /sys-analyst/
Disallow: /sys-architect/
# 禁止抓取静态资源目录（无 SEO 价值）
Disallow: /static/
# 禁止抓取标签页面（避免重复内容）
Disallow: /tags/
# 禁止抓取 Clippings 目录
Disallow: /Clippings/

# -------------------- 主流搜索引擎特殊规则 --------------------

# Google：遵守 robots.txt，允许适度抓取
User-agent: Googlebot
Allow: /
Disallow: /sys-analyst/
Disallow: /sys-architect/
Disallow: /static/
Disallow: /tags/
Disallow: /Clippings/
Crawl-delay: 1

# Bing
User-agent: Bingbot

Disallow: /sys-analyst/
Disallow: /sys-architect/
Disallow: /static/
Disallow: /tags/
Disallow: /Clippings/
